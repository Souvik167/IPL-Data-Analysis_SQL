SELECT DISTINCT bowler, 
	   COUNT(bowler) AS balls, 
		(COUNT(bowler)/6) || '.' || (COUNT(bowler)%6) AS Over, 
		SUM(total_runs) AS run, 
		ROUND(SUM(total_runs)/(COUNT(bowler)/6.0),2) AS economy
FROM ipl_balls 
GROUP BY DISTINCT bowler 
HAVING COUNT(bowler)>500 
ORDER BY economy 
LIMIT 10;
